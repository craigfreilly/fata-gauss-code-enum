\begin{frame}{What is a knot?}
    \begin{itemize}[<+->]
        \item A \emph{knot} is an embedding of the circle in $\mathbb{R}^3$.
        \item An intuitive way to think about this is to consider a knot as a knotted piece of string with the ends glued together.
    \end{itemize}
\end{frame}

\begin{frame}{Representations of knots}
    \only<1>{
    \begin{itemize}
        \item Knot diagrams are really just 4-valent planar graphs.
            \begin{itemize}
                \item The vertices in the graph correspond to the crossings in the knot diagram.
                \item The arcs between vertices correspond to arcs between crossings in the knot diagram.
                \item The arcs are decorated with their orientation at their source and target crossings.
            \end{itemize}
        \item Other data structures familiar to computer scientists can be used, linked lists of crossings were popular in the 1950's.
    \end{itemize}
    }
    \only<2>
    {
    \begin{itemize}
        \item The representations used by topologists are typically also used for representing knots in a computer.
        \item Examples are Dowker-Thistlethwait codes (DT codes), \textbf{Gauss codes}, braid representatives, Conway notation, and many more.
    \end{itemize}
    }
\end{frame}

\begin{frame}{Knot enumeration by Gauss codes}
	\begin{itemize}
		\item There are three sensible questions to ask:
		\begin{itemize}
			\item Do all knots give rise to a Gauss code? \only<2->{Yes.}\only<3->{ Trivially.}
			\item Can two Gauss codes correspond to the same knot? \only<4->{Yes.}
			\item Do all Gauss codes represent a knot? \only<5->{No!}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Symmetries involved in Gauss code enumeration}
	\only<1>{
	\begin{columns}
        \begin{column}{0.65\textwidth}
			\begin{itemize}
				\item The choices involved in each stage of obtaining a Gauss code from a given knot lead to rich symmetries in the problem of enumerating knots by Gauss codes.
				\item As we have seen many Gauss codes represent the same knot.
			\end{itemize}
		\end{column}
		\begin{column}{0.25\textwidth}
		\only<2>{
			\begin{tikzpicture}
				\foreach \brk in {0,1,2} {
				\begin{scope}[rotate=\brk * 120]
				\node[knot crossing, transform shape, inner sep=1.5pt] (k\brk) at (0,-1) {};
				\end{scope}
				}
				\foreach \brk in {0,1,2} {
				\pgfmathparse{int(Mod(\brk - 1,3))}
				\edef\brl{\pgfmathresult}
				\draw[ultra thick,red] (k\brk) .. controls (k\brk.4 north west) and (k\brl.4 north east) .. (k\brl.center);

		\end{column}
	\end{columns}
	}
	\only<3->{
		\begin{itemize}
			\item The group of the symmetries of the problem is $S_{2n} \times Z_{2n} \times Z_2$ (over some quotient).
			\item Symmetry breaking on each of these groups is easy, but this is not the case for their product.
		\end{itemize}
	}
\end{frame}

\begin{frame}[fragile]{The dually pairedness condition}
    \only<-2>{
    \begin{itemize}
        \item The other necessary condition, which when combined with the evenly spaced condition is sufficient,
            is that, for a Gauss code $w$ its associated $w^{\ast}$  must be dually paired.
        \item<2> For a Gauss code $w$, its associated $w^{\ast}$ is obtained by the following algorithm: for
each letter i in the code, reverse the order of all the letters between each instance of i in turn.
    \end{itemize}
    }
    \only<3-6>{
    \begin{description}
	\item[Original code]<3-> 1, 2, 3, 1, 4, 3, 2, 4
	\item[Order between 1 reversed]<3-6>1, 3, 2, 1, 4, 3, 2, 4
	\item[Order between 2 reversed]<4-6>1, 3, 2, 3, 4, 1, 2, 4
	\item[Order between 3 reversed]<5-6>1, 3, 2, 3, 4, 1, 2, 4
	\item[Order between 4 reversed]<6>1, 3, 2, 3, 4, 2, 1, 4
    \end{description}
    }   
    \only<7->{
     \begin{columns}
         \begin{column}{0.70\textwidth}
             \begin{itemize}
                 \item For a shadow Gauss code $w$ two labels $i$ and $j$ are called \emph{interlaced} in $w$ iff \newline $w = w_1 \text{ } i \text{ } w_2 \text{ } j \text{ } w_3 \text{ } i \text{ } w_4 \text{ } j \text{ } w_5$ or $w = w_1 \text{ } j \text{ } w_2 \text{ } i \text{ } w_3 \text{ } j \text{ } w_4 \text{ } i \text{ } w_5$, \newline where each $w_n$ is a subcode (which is possibly empty). 
                 \item An unsigned Gauss code's $w^{\ast}$ construction is \emph{dually paired} iff each letter $i \in$ $w^{\ast}$ can be placed into one of two sets, such that each $i$ in both sets is not interlaced with another letter contained in the same set as $i$.
             \end{itemize}
         \end{column}

\begin{frame}{Implementation}
    \only<1>{
        \begin{itemize}
            \item The aim of the project was to generate all knot diagrams of a given crossings number, up to some notion of equivalence.
            \item The project made use of both Constraint Programming and dedicated algorithms to efficently filter codes.
        \end{itemize}
    }
    \only<2->{
        \begin{itemize}
            \item Constraint programming:
                \begin{itemize}
                    \item Generate Gauss codes which pass the evenly spaced condition.
                    \item Symmetry breaking on the $S_{n}$ symmetry.
                \end{itemize}
            \item Filtering:
                \begin{itemize}
                    \item Test that the code is dually paired.
                    \item Test that the code is the lexicographically minium code in its orbit in the group $\mathbb{Z}_{2n} \times \mathbb{Z}_2$.
                \end{itemize}
        \end{itemize}
    }
\end{frame}
